# Combobox

## Usage

```tsx
function Component() {
  return (
    <Combobox>
      <ComboboxControl>
        <ComboboxInput />
        <ComboboxArrow />
      </ComboboxControl>

      <ComboboxOptions>
        <ComboboxOption label="Option 1" value="1" />
        <ComboboxOption label="Option 2" value="2" />
        <ComboboxOption label="Option 3" value="3" />
      </ComboboxOptions>
    </Combobox>
  );
}
```
